bool isAdtsAac(Uint8List data) {
  return data.length >= 7 &&
      data[0] == 0xFF &&
      (data[1] & 0xF0) == 0xF0;
}

bool looksLikeRtpMpeg4GenericAac(Uint8List data) {
  return data.length >= 4 &&
      data[0] == 0x00 &&
      data[1] == 0x10;
}

Uint8List prepareAacData(AudioStreamInfo info, Uint8List current) {
  // Already ADTS AAC, so send as-is.
  if (isAdtsAac(current)) {
    return current;
  }

  // Your current getAacData() appears to handle this packet shape:
  // 00 10 | AU header | raw AAC
  if (looksLikeRtpMpeg4GenericAac(current)) {
    final rawAac = getAacData(current);
    if (rawAac.isEmpty) return rawAac;

    final header = getAdtsHeader(info, rawAac.length);

    return Uint8List.fromList([
      ...header,
      ...rawAac,
    ]);
  }

  // Unknown / maybe already raw AAC.
  // This may or may not work depending on what the bridge decoder expects.
  return current;
}
